\bigskip

\IncMargin{2em}
\begin{algorithm}\label{algoritmus1}
\caption{\textsc{FastSLAM}}
\SetNlSty{}{}{:}
\SetAlgoNoLine
\SetAlgoLongEnd
\Indm\Indmm
\KwIn{$(X_{t-1},u_t,z_t)$}
\KwOut{$X_t$}
\BlankLine
\Indp\Indpp
	$\overline{X_t} = X_t = 0$\\
	\For{$k = 1$ to $M$}
	{$x_t^{[k]} = sample\_motion\_model(u_t,{x^{[k]}_{t-1}})$\\
	$\omega_t^{[k]} = measurement\_model(z_t,{x_t}^{[k]},m_{t-1})$\\
	$m_t^{[k]} = updated\_occupancy\_grid(z_t,{x_t}^{[k]},m^{[k]}_{t-1})$\\
	$\overline{X_t}=\overline{X_t}+\langle x^{[m]}_x,\omega^{[m]}_t\rangle $}
	\For{$k = 1$ to $M$}
	{draw $i$ with probability $\approx \omega^{[i]}_t$\\
	add	$\langle x^{[k]}_x,m^{[k]}_t\rangle$ to $X_t$}
	\KwRet{$X_t$}\\
\end{algorithm}
